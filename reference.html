function addClass( classname, element ) {
    var cn = element.className;
    //test for existance
    if( cn == "undefined" ) {
        return;
    }
    //add a space if the element already has class
    if( cn != '' ) {
        classname = ' ' + classname;
    }
    element.className = cn + classname;
}
